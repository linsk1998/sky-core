queueMicrotask

在浏览器完成后面的其他语句后,就立刻执行这个回调函数。多层queueMicrotask在setTimeout 0之前执行。

setTimeout(function(){alert(2)},0);
queueMicrotask(function(){
	queueMicrotask(function(){
		queueMicrotask(function(){
			queueMicrotask(function(){
				queueMicrotask(function(){
					queueMicrotask(function(){
						queueMicrotask(function(){
							alert(1);
						});
					});
				});
			});
		});
	});
});
queueMicrotask(function(){
	alert(0);
});
